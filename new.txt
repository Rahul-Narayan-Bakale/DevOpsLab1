# Set Git username and email
git config --global user.name "Prajwal Biradar"
git config --global user.email "your-email@example.com"

# Go to Desktop
cd Desktop

# Create a folder
mkdir pr1

# Enter the folder
cd pr1

# Initialize Git repository
git init

# Create a file
echo "hello" > first.txt

# Check status
git status

# Add file to staging
git add .

# Commit the file
git commit -m "Initial Commit"

# Add remote GitHub repository
git remote add origin https://github.com/Prajwalgb99/p1.git

# Ensure branch name is main
git branch -M main

# Push to GitHub
git push -u origin main

# Go back to Desktop
cd ..

# Clone the repository from GitHub
git clone https://github.com/Prajwalgb99/p1.git

# Enter cloned repository
cd p1

# Check commits
git log

# Reset last commit (only works if more than one commit exists)
git reset --soft HEAD~1

# Configure Git username and email
git config --global user.name "Prajwal Biradar"
git config --global user.email "your-email@example.com"

# Go to Desktop
cd Desktop

# Clone the empty repository
git clone https://github.com/Prajwalgb99/p2.git

# Enter the repository
cd p2

# Create a file
echo "hello" > first.txt

# Add file
git add .

# First commit
git commit -m "Initial Commit"

# Create a new branch
git checkout -b feature

# Modify file in feature branch
echo "hi" > first.txt

# Add changes
git add .

# Commit changes
git commit -m "Second Commit in feature branch"

# Switch back to main branch
git checkout main

# Modify file in main branch
echo "hello hi" > first.txt

# Add changes
git add first.txt

# Commit changes
git commit -m "Third Commit in main branch"

# Push main branch to GitHub
git push origin main

# Show difference between branches
git diff main feature

# Merge feature branch into main
git merge feature

# (If conflict occurs, edit file manually then run:)
git add .

# Commit merge resolution
git commit -m "Resolve Conflict"

# Push final code
git push origin main
